### MDM 配置来源

```typescript
// macOS: ~/Library/Managed Preferences/com.anthropic.claudecode.plist
// Windows: HKLM\SOFTWARE\Policies\Anthropic\ClaudeCode
// Linux: /etc/claudecode/mdm.conf

async function loadMDMConfig(): MDMConfig | null {
  // macOS plist
  if (platform === 'darwin') {
    const plistPath = '~/Library/Managed Preferences/com.anthropic.claudecode.plist'
    if (fs.existsSync(plistPath)) {
      return parsePlist(await fs.readFile(plistPath))
    }
  }

  // Windows registry
  if (platform === 'win32') {
    const registryKey = 'HKLM\\SOFTWARE\\Policies\\Anthropic\\ClaudeCode'
    return readRegistry(registryKey)
  }

  // Linux conf
  if (platform === 'linux') {
    const confPath = '/etc/claudecode/mdm.conf'
    if (fs.existsSync(confPath)) {
      return parseConf(await fs.readFile(confPath))
    }
  }

  return null  // 无 MDM 配置
}
```

## MDM 优先级

### 配置优先级

```typescript
// 配置优先级：MDM > 用户设置 > 默认值

function resolveConfig(
  mdmConfig: MDMConfig | null,
  userConfig: UserConfig,
  defaultConfig: DefaultConfig
): ResolvedConfig {
  // MDM 配置优先
  if (mdmConfig) {
    return {
      allowedMcpServers: mdmConfig.allowedMcpServers || defaultConfig.allowedMcpServers,
      deniedMcpServers: mdmConfig.deniedMcpServers || [],
      allowedTools: mdmConfig.allowedTools || defaultConfig.allowedTools,
      deniedTools: mdmConfig.deniedTools || [],
      permissionMode: mdmConfig.defaultPermissionMode || 'plan',
      sandboxLevel: mdmConfig.sandboxLevel || 'none',
      auditLogPath: mdmConfig.auditLogPath || defaultConfig.auditLogPath,
      telemetryEnabled: mdmConfig.telemetryEnabled !== false,  // 默认启用
    }
  }

  // 用户配置
  return {
    ...defaultConfig,
    ...userConfig,
  }
}
```

### MDM 强制性

```typescript
// MDM 配置不可被用户覆盖

function checkMDMOverride(key: string): boolean {
  // MDM 强制字段，用户无法修改
  const MDM_ENFORCED_FIELDS = [
    'allowedMcpServers',
    'deniedMcpServers',
    'deniedTools',
    'auditLogPath',
    'telemetryEnabled',
  ]

  return MDM_ENFORCED_FIELDS.includes(key)
}

async function saveUserConfig(key: string, value: any): void {
  // 检查 MDM 强制
  const mdmConfig = await loadMDMConfig()

  if (mdmConfig && checkMDMOverride(key)) {
    throw new Error(`配置 '${key}' 由企业 MDM 管理，无法修改`)
  }

  // 用户配置保存
  userSettings[key] = value
  await saveUserSettings()
}
```

```typescript
interface AuditLogEntry {
  timestamp: string           // ISO 时间戳
  sessionId: string           // 会话 ID
  userId: string              // 用户 ID（如果可用）
  eventType: AuditEventType   // 事件类型
  toolName?: string           // 工具名称
  toolInput?: object          // 工具输入（过滤敏感信息）
  toolOutput?: string         // 工具输出摘要
  permissionDecision?: PermissionDecision  // 权限决策
  reasoning?: string          // 决策原因
  userResponse?: UserResponse  // 用户响应
}

type AuditEventType =
  | 'tool_execution'
  | 'permission_request'
  | 'permission_granted'
  | 'permission_denied'
  | 'session_start'
  | 'session_end'
  | 'mcp_connection'
  | 'mcp_tool_call'
  | 'file_operation'
  | 'command_execution'
```

### 日志写入

```typescript
class AuditLogger {
  private logPath: string
  private writeQueue: AuditLogEntry[] = []

  constructor(logPath: string) {
    this.logPath = logPath
    this.startFlushLoop()
  }

  log(entry: AuditLogEntry): void {
    // 添加到队列
    this.writeQueue.push(entry)
  }

  private async flush(): void {
    if (this.writeQueue.length === 0) return

    // 批量写入
    const entries = this.writeQueue.splice(0, this.writeQueue.length)
    const lines = entries.map(e => JSON.stringify(e) + '\n').join('')

    await fs.appendFile(this.logPath, lines, 'utf-8')
  }

  private startFlushLoop(): void {
    // 每 5 秒刷新一次
    setInterval(() => this.flush(), 5000)
  }
}
```

### 工具执行日志

```typescript
async function logToolExecution(
  toolName: string,
  input: object,
  output: ToolResult,
  decision: PermissionDecision
): void {
  const entry: AuditLogEntry = {
    timestamp: new Date().toISOString(),
    sessionId: context.sessionId,
    userId: getUserId(),
    eventType: 'tool_execution',
    toolName,
    toolInput: sanitizeInput(input),  // 过滤敏感信息
    toolOutput: summarizeOutput(output),
    permissionDecision: decision,
  }

  auditLogger.log(entry)
}

// 过滤敏感信息
function sanitizeInput(input: object): object {
  const sanitized = { ...input }

  const sensitiveFields = ['password', 'token', 'apiKey', 'secret', 'key']
  for (const field of sensitiveFields) {
    if (sanitized[field]) {
      sanitized[field] = '[REDACTED]'
    }
  }

  return sanitized
}

// 输出摘要
function summarizeOutput(output: ToolResult): string {
  if (output.success) {
    // 成功输出摘要
    return `Success (${output.type || 'unknown'})`
  } else {
    // 错误输出
    return `Failed: ${output.error}`
  }
}
```

## 企业合规

### SOC 2 合规

```typescript
// SOC 2 要求
const SOC2_REQUIREMENTS = {
  // 访问控制
  accessControl: {
    uniqueIds: true,           // 每个用户有唯一 ID
    roleBasedAccess: true,     // 基于角色的访问
    leastPrivilege: true,      // 最小权限原则
  },

  // 审计日志
  audit: {
    logAllAccess: true,        // 记录所有访问
    logRetention: 365,         // 日志保留 365 天
    logIntegrity: true,        // 日志完整性
  },

  // 数据保护
  dataProtection: {
    encryptionAtRest: true,    // 静态加密
    encryptionInTransit: true, // 传输加密
    dataRetention: true,       // 数据保留策略
  },
}

async function checkSOC2Compliance(): ComplianceReport {
  const checks = []

  // 检查审计日志
  const auditLogPath = getAuditLogPath()
  if (!auditLogPath) {
    checks.push({ requirement: 'audit.logAllAccess', compliant: false })
  } else {
    checks.push({ requirement: 'audit.logAllAccess', compliant: true })
  }

  // 检查权限控制
  const permissionMode = getPermissionMode()
  if (permissionMode !== 'auto') {
    checks.push({ requirement: 'accessControl.leastPrivilege', compliant: true })
  } else {
    checks.push({ requirement: 'accessControl.leastPrivilege', compliant: false })
  }

  return {
    standard: 'SOC2',
    checks,
    overall: checks.every(c => c.compliant),
  }
}
```

```typescript
// GDPR 要求
const GDPR_REQUIREMENTS = {

  dataMinimization: {

  },

  },

async function handleGDPRRequest(
  userId: string,
  requestType: 'access' | 'delete' | 'export'
): GDPRResponse {
  switch (requestType) {
    case 'access':
      // 返回用户所有数据
      return exportUserData(userId)

    case 'delete':
      // 删除用户所有数据
      await deleteUserData(userId)
      return { success: true }

    case 'export':
      // 导出用户数据（可携带）
      return exportUserData(userId)
  }
}
```
