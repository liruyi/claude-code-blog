## 大文件处理

### 分块读取

```typescript
const MAX_READ_SIZE = 100000  // 100KB

async function readFile(path: string): ReadResult {
  const stats = await fs.stat(path)

  if (stats.size > MAX_READ_SIZE) {
    // 大文件分块读取
    return readLargeFile(path, stats.size)
  }

  // 小文件完整读取
  const content = await fs.readFile(path, 'utf-8')
  return { content, isPartial: false }
}

async function readLargeFile(path: string, size: number): ReadResult {
  // 只读取前 MAX_READ_SIZE
  const fd = await fs.open(path, 'r')
  const buffer = Buffer.alloc(MAX_READ_SIZE)
  await fd.read(buffer, 0, MAX_READ_SIZE, 0)
  await fd.close()

  const content = buffer.toString('utf-8')

  return {
    content,
    isPartial: true,
    totalSize: size,
    readSize: MAX_READ_SIZE,
    message: `文件过大，只显示前 ${MAX_READ_SIZE} 字节`,
  }
}
```

## 工具输出截断

### 输出限制

```typescript
const MAX_OUTPUT_SIZE = 50000  // 50KB

async function truncateOutput(output: string): TruncatedOutput {
  if (output.length <= MAX_OUTPUT_SIZE) {
    return { content: output, truncated: false }
  }

  // 截断并添加摘要
  const truncated = output.slice(0, MAX_OUTPUT_SIZE)
  const summary = await summarizeOutput(output)

  return {
    content: truncated,
    truncated: true,
    originalSize: output.length,
    summary,
    message: `输出过长，已截断至 ${MAX_OUTPUT_SIZE} 字节`,
  }
}
```

### Bash 输出截断

```typescript
async function bashTool(input: BashInput): BashResult {
  const result = await executeCommand(input.command)

  // 检查输出大小
  if (result.stdout.length > MAX_OUTPUT_SIZE) {
    // 截断 stdout
    const truncated = await truncateOutput(result.stdout)

    return {
      stdout: truncated.content,
      stderr: result.stderr,
      truncated: true,
      summary: truncated.summary,
    }
  }

  return result
}
```

### Grep 结果限制

```typescript
const MAX_MATCHES = 250

async function grepTool(input: GrepInput): GrepResult {
  const matches = await searchFiles(input.pattern, input.path)

  if (matches.length > MAX_MATCHES) {
    // 限制匹配数量
    const limitedMatches = matches.slice(0, MAX_MATCHES)

    return {
      matches: limitedMatches,
      totalMatches: matches.length,
      limited: true,
      message: `找到 ${matches.length} 个匹配，只显示前 ${MAX_MATCHES} 个`,
    }
  }

  return { matches, totalMatches: matches.length, limited: false }
}
```

## 消息生命周期

### 消息状态

```typescript
type MessageState = 'active' | 'compacted' | 'archived'

interface Message {
  id: string
  state: MessageState
  content: string
  tokenCount: number
  createdAt: number
  lastAccessedAt: number
}
```

### 活跃消息管理

```typescript
const MAX_ACTIVE_TOKENS = 150000

function manageActiveMessages(messages: Message[]): Message[] {
  // 计算活跃消息总 token
  const activeMessages = messages.filter(m => m.state === 'active')
  const totalTokens = activeMessages.reduce((sum, m) => sum + m.tokenCount, 0)

  if (totalTokens > MAX_ACTIVE_TOKENS) {
    // 压缩旧消息
    const oldMessages = activeMessages.slice(0, -10)
    for (const msg of oldMessages) {
      msg.state = 'compacted'
    }

    // 生成压缩摘要
    const summary = generateCompactSummary(oldMessages)
    messages.push({
      id: generateId(),
      state: 'active',
      content: summary,
      tokenCount: estimateTokens(summary),
      createdAt: Date.now(),
      lastAccessedAt: Date.now(),
    })
  }

  return messages
}
```

### 消息清理

```typescript
const ARCHIVE_THRESHOLD = 7 * 24 * 60 * 60 * 1000  // 7 天

function cleanupOldMessages(messages: Message[]): Message[] {
  const now = Date.now()

  for (const msg of messages) {
    if (msg.state === 'compacted' && now - msg.createdAt > ARCHIVE_THRESHOLD) {
      // 7 天以上的压缩消息归档
      msg.state = 'archived'
    }
  }

  // 移除归档消息内容（只保留元数据）
  const archived = messages.filter(m => m.state === 'archived')
  for (const msg of archived) {
    // 只保留摘要，删除完整内容
    msg.content = msg.summary || ''
    msg.tokenCount = estimateTokens(msg.content)
  }

  return messages
}
```

## 缓存清理

### 文件缓存限制

```typescript
const MAX_FILE_CACHE_SIZE = 10  // 最多缓存 10 个文件
const MAX_CACHE_MEMORY = 500000  // 500KB

function cleanupFileCache(): void {
  const cacheEntries = [...readFileState.entries()]

  // 检查缓存数量
  if (cacheEntries.length > MAX_FILE_CACHE_SIZE) {
    // 移除最旧的
    const sorted = cacheEntries.sort((a, b) =>
      a[1].lastReadAt - b[1].lastReadAt
    )

    const toRemove = sorted.slice(0, cacheEntries.length - MAX_FILE_CACHE_SIZE)
    for (const [path] of toRemove) {
      readFileState.delete(path)
    }
  }

  // 检查缓存内存
  const totalMemory = cacheEntries.reduce((sum, [, state]) =>
    sum + state.content.length, 0
  )

  if (totalMemory > MAX_CACHE_MEMORY) {
    // 清理大文件缓存
    const largeFiles = cacheEntries.filter(([, state]) =>
      state.content.length > MAX_CACHE_MEMORY / MAX_FILE_CACHE_SIZE
    )

    for (const [path] of largeFiles) {
      readFileState.delete(path)
    }
  }
}
```

```typescript
const MEMORY_WARNING_THRESHOLD = 500 * 1024 * 1024  // 500MB

function checkMemoryWarning(): void {
  const stats = getMemoryStats()

  if (stats.totalHeapUsed > MEMORY_WARNING_THRESHOLD) {
    // 内存警告
    console.warn(`内存使用过高: ${stats.totalHeapUsed / 1024 / 1024}MB`)

    // 触发清理
    cleanupFileCache()
    cleanupOldMessages(messages)

    // 考虑垃圾回收
    if (global.gc) {
      global.gc()
    }
  }
}
```

```typescript
function snipCompact(messages: Message[]): Message[] {
  // Snip 压缩：只保留摘要
  const snipped = messages.map(msg => {
    if (msg.type === 'tool_result' && msg.output.length > 1000) {
      return {
        ...msg,
        output: '',  // 清空输出
        summary: summarizeOutput(msg.output),
        tokenCount: 100,  // 大幅减少

    return msg

  return snipped
}
```

## 内存原则总结

### 1. 大文件分块

```
超过 100KB → 分块读取 → 提供选择
```

### 2. 输出截断

```
超过 50KB → 截断 + 摘要 → 减少 token
```

### 3. 消息状态管理

```
active → compacted → archived → 清理
```

### 4. 缓存限制

```
最多 10 个文件，500KB 内存
```

### 5. 定期清理

```
每 5 分钟检查，超过阈值清理
```
