AI CLI 不只是文本输入。本篇分析 Claude Code 的多模态输入设计。

## 多模态输入的必要性

### 用户场景

```
场景1：代码截图分析
用户：截图一个错误 → 粘贴 → AI 分析
效果：AI 直接看到错误，不需要描述

场景2：文件快速添加
用户：拖拽配置文件 → AI 理解上下文
效果：避免手动输入路径

场景3：设计稿讨论
用户：粘贴 UI 设计稿 → AI 提建议
效果：视觉沟通更直观

场景4：语音快速提问
用户：语音描述需求 → AI 理解
效果：解放双手，快速交互
```

### 拖拽检测

```typescript
// 终端不支持真正的拖拽，但可以通过 CLI 参数
// claude @file1 @file2 @image.png

function detectFileAttachments(args: string[]): FileAttachment[] {
  const attachments: FileAttachment[] = []

  for (const arg of args) {
    if (arg.startsWith('@')) {
      const path = arg.slice(1)

      // 检查文件是否存在
      if (fs.existsSync(path)) {
        const stats = fs.statSync(path)
        const mimeType = detectMimeType(path)

        attachments.push({
          type: 'file',
          path,
          size: stats.size,
          mimeType,
        })
      }
    }
  }

  return attachments
}
```

### 文件附件处理

```typescript
async function processFileAttachment(attachment: FileAttachment): ProcessedAttachment {
  const { path, mimeType } = attachment

  // 图片文件
  if (mimeType.startsWith('image/')) {
    const base64 = await fs.readFile(path, 'base64')
    return {
      type: 'image',
      source: {
        type: 'base64',
        media_type: mimeType,
        data: base64,
      },
      metadata: {
        path,
        size: attachment.size,
      },
    }
  }

  // PDF 文件
  if (mimeType === 'application/pdf') {
    return {
      type: 'document',
      source: { type: 'file', path },
      metadata: {
        path,
        pages: await countPdfPages(path),
      },
    }
  }

  // 文本文件
  if (isTextFile(path)) {
    const content = await fs.readFile(path, 'utf-8')
    return {
      type: 'text',
      content,
      metadata: {
        path,
        lines: content.split('\n').length,
      },
    }
  }

  // 其他文件
  return {
    type: 'binary',
    source: { type: 'file', path },
    metadata: {
      path,
      mimeType,
      size: attachment.size,
    },
  }
}
```

### 文件大小限制

```typescript
const FILE_SIZE_LIMITS = {
  image: 10 * 1024 * 1024,      // 10MB
  pdf: 20 * 1024 * 1024,        // 20MB
  text: 100 * 1024,             // 100KB
  binary: 50 * 1024 * 1024,     // 50MB
}

function checkFileSizeLimit(attachment: FileAttachment): boolean {
  const limit = FILE_SIZE_LIMITS[attachment.mimeType?.split('/')[0]] ||
                FILE_SIZE_LIMITS.binary

  if (attachment.size > limit) {
    return false
  }

  return true
}
```

## 图片粘贴

### 图片检测

```typescript
// 从剪贴板检测图片
async function detectClipboardImage(): ImageAttachment | null {
  // macOS: 使用 pngpaste
  // Linux: 使用 xclip
  // Windows: 使用 PowerShell

  if (platform === 'darwin') {
    const result = await spawn('pngpaste -')
    if (result.success) {
      return {
        type: 'image',
        content: result.stdout,  // base64
        mimeType: 'image/png',
      }
    }
  }

  if (platform === 'linux') {
    const result = await spawn('xclip -selection clipboard -t image/png -o')
    if (result.success) {
      return {
        type: 'image',
        content: result.stdout.toString('base64'),
        mimeType: 'image/png',
      }
    }
  }

  if (platform === 'win32') {
    const script = `
      Add-Type -AssemblyName System.Windows.Forms
      $img = [Windows.Forms.Clipboard]::GetImage()
      if ($img) {
        $ms = New-Object System.IO.MemoryStream
        $img.Save($ms, [System.Drawing.Imaging.ImageFormat]::Png)
        [Convert]::ToBase64String($ms.ToArray())
      }
    `
    const result = await spawn(`powershell -command '${script}'`)
    if (result.success) {
      return {
        type: 'image',
        content: result.stdout,
        mimeType: 'image/png',
      }
    }
  }

  return null
}
```

```typescript
// 检测粘贴快捷键
function setupPasteHandler(inputBox: InputBox): void {
  // 监听 Ctrl+V (Windows/Linux) 或 Cmd+V (macOS)
  inputBox.on('paste', async () => {
    // 先检测图片
    const image = await detectClipboardImage()

    if (image) {
      // 有图片，添加到附件
      addImageAttachment(image)

      // 显示提示
      showToast('图片已添加到附件')

      // 不继续文本粘贴流程
      return
    }

    // 无图片，正常文本粘贴
    const text = await detectClipboardText()
    if (text) {
      inputBox.insertText(text)
    }
  })
}
```

### 图片处理

```typescript
async function processImageAttachment(image: ImageAttachment): ProcessedImage {
  // 1. 检查大小
  const size = Buffer.byteLength(image.content, 'base64')
  if (size > FILE_SIZE_LIMITS.image) {
    // 压缩图片
    const compressed = await compressImage(image.content)
    return {
      source: {
        type: 'base64',
        media_type: compressed.mimeType,
        data: compressed.data,
      },
      metadata: {
        originalSize: size,
        compressedSize: compressed.size,
        compressed: true,
      },
    }
  }

  // 2. 检查格式
  const supportedFormats = ['image/png', 'image/jpeg', 'image/gif', 'image/webp']
  if (!supportedFormats.includes(image.mimeType)) {
    // 转换格式
    const converted = await convertImage(image.content, 'image/png')
    return {
      source: {
        type: 'base64',
        media_type: 'image/png',
        data: converted,
      },
      metadata: {
        converted: true,
        originalFormat: image.mimeType,
      },
    }
  }

  // 3. 返回原始
  return {
    source: {
      type: 'base64',
      media_type: image.mimeType,
      data: image.content,
    },
    metadata: {
      size,
    },
  }
}
```

```typescript
async function compressImage(base64: string): CompressedImage {
  // 使用 image processor (vendor 模块)
  const processor = new ImageProcessor()

  const result = await processor.compress(base64, {
    maxSize: 5 * 1024 * 1024,  // 5MB
    quality: 0.8,
  })

  return {
    data: result.base64,
    mimeType: result.mimeType,
    size: result.size,
  }
}
```

## 消息构建

### 多模态消息结构

```typescript
interface MultimodalMessage {
  role: 'user'
  content: ContentBlock[]
}

type ContentBlock =
  | { type: 'text', text: string }
  | { type: 'image', source: ImageSource }
  | { type: 'document', source: DocumentSource }

function buildMultimodalMessage(
  text: string,
  attachments: ProcessedAttachment[]
): MultimodalMessage {
  const content: ContentBlock[] = []

  // 文本
  if (text) {
    content.push({ type: 'text', text })
  }

  // 图片
  for (const att of attachments) {
    if (att.type === 'image') {
      content.push({
        type: 'image',
        source: att.source,
      })
    }
  }

  // 文档
  for (const att of attachments) {
    if (att.type === 'document' || att.type === 'text') {
      content.push({
        type: 'document',
        source: {
          type: 'text',
          text: att.content,
          title: att.metadata.path,
        },
      })
    }
  }

  return { role: 'user', content }
}
```

## 附件管理

### 附件列表

```typescript
interface AttachmentManager {
  attachments: Attachment[]
  add(attachment: Attachment): void
  remove(index: number): void
  clear(): void
  getTotalSize(): number
  validate(): ValidationResult
}

const manager: AttachmentManager = {
  attachments: [],

  add(attachment) {
    // 检查限制
    if (this.attachments.length >= MAX_ATTACHMENTS) {
      throw new Error('附件数量超过限制')
    }

    if (this.getTotalSize() + attachment.size > MAX_TOTAL_SIZE) {
      throw new Error('附件总大小超过限制')
    }

    this.attachments.push(attachment)
  },

  remove(index) {
    this.attachments.splice(index, 1)
  },

  clear() {
    this.attachments = []
  },

  getTotalSize() {
    return this.attachments.reduce((sum, a) => sum + a.size, 0)
  },

  validate() {
    const errors = []

    // 检查数量
    if (this.attachments.length > MAX_ATTACHMENTS) {
      errors.push('附件数量超过限制')
    }

    // 检查大小
    if (this.getTotalSize() > MAX_TOTAL_SIZE) {
      errors.push('附件总大小超过限制')
    }

    // 检查格式
    for (const att of this.attachments) {
      if (!isSupportedFormat(att.mimeType)) {
        errors.push(`不支持的格式: ${att.mimeType}`)
      }
    }

    return { valid: errors.length === 0, errors }
  },
}
```

## 多模态原则总结

### 1. 自然集成
